  1: { seeds: 2000, frames: 500, startFrame: 0, startSeed: 0 },
  2: { seeds: 4000, frames: 550, startFrame: 500, startSeed: 2000 },
  3: { seeds: 6000, frames: 570, startFrame: 1050, startSeed: 6000 },
  4: { seeds: 8000, frames: 650, startFrame: 1620, startSeed: 12000 },
  5: { seeds: 10000, frames: 750, startFrame: 2270, startSeed: 20000 },
  6: { seeds: 30000, frames: 4385, startFrame: 3020, startSeed: 30000 }
};

function mapSeedsToFrame(seedCount, sceneNumber) {
  const config = SCENE_CONFIG[sceneNumber];
  if (!config) return 0;
  
  const sceneSeeds = Math.min(
    Math.max(0, seedCount - config.startSeed),
    config.seeds
  );
  
  const progress = sceneSeeds / config.seeds;
  return config.startFrame + Math.floor(progress * config.frames);
}

// ARQUIVO: wallet-connector.js

function getBitcoinProvider() {
  if (window.unisat) return window.unisat;
  if (window.xverse) return window.xverse;
  if (window.btc) return window.btc;
  if (window.leather) return window.leather;
  return null;
}

async function connectWallet() {
  const provider = getBitcoinProvider();
  if (!provider) {
    alert('Instale Unisat, Xverse ou Leather para participar');
    return null;
  }
  
  try {
    const accounts = await provider.requestAccounts();
    const address = accounts[0];
    const pubkey = await provider.getPublicKey();
    
    localStorage.setItem('bbp_wallet_address', address);
    localStorage.setItem('bbp_wallet_pubkey', pubkey);
    
    updateWalletUI(address);
    return { address, pubkey, provider };
  } catch (error) {
    console.error('Erro ao conectar:', error);
    return null;
  }
}

const BASE_REWARD = 10;

async function grantRewards(address, sceneNumber, position) {
  let fragments = BASE_REWARD;
  
  // Bônus para primeiros 100
  if (position <= 100) fragments += 50;
  
  // Bônus por compartilhamentos
  const shares = await getShareCount(address);
  fragments += Math.min(shares, 3) * 5;
  
  // Registra na rede P2P
  const userFragments = gun.get('bbp_ritual').get('protocol_fragments').get(address);
  userFragments.once((data) => {
    const current = data?.amount || 0;
    userFragments.put({
      amount: current + fragments,
      last_updated: Date.now(),
      history: [...(data?.history || []), {
        amount: fragments,
        scene: sceneNumber,
        timestamp: Date.now()
      }]
    });
  });
  
  // Concede fração do NFT da cena (se disponível)
  await grantSceneFraction(address, sceneNumber);
  
  showRewardNotification(fragments, sceneNumber);
}

function checkAndRedirect() {
  const currentScene = getCurrentSceneBySeedCount(currentSeedCount);
  const currentPath = window.location.pathname;

    localStorage.setItem('bbp_seed_count', currentSeedCount);
    window.location.href = expectedPath;
  }
}

// ARQUIVO: multi-chain-indexer.js

const SUPPORTED_NETWORKS = [
  'bitcoin', 'ethereum', 'polygon', 'solana',
  'arbitrum', 'optimism', 'base', 'avalanche', 'bsc'
];

const NETWORK_ENDPOINTS = {
  bitcoin: 'https://mempool.space/api/address',
  ethereum: 'https://api.etherscan.io/api',
  polygon: 'https://api.polygonscan.com/api',
  solana: 'https://api.solscan.io/account',
  arbitrum: 'https://api.arbiscan.io/api',
  optimism: 'https://api-optimistic.etherscan.io/api',
  base: 'https://api.basescan.org/api',
  avalanche: 'https://api.snowtrace.io/api',
  bsc: 'https://api.bscscan.com/api'
};

async function getAssetStateAcrossNetworks(assetId) {
  const states = {};
  
  for (const network of SUPPORTED_NETWORKS) {
    try {
      const response = await fetch(`${NETWORK_ENDPOINTS[network]}/${assetId}`);
      states[network] = await response.json();
    } catch (error) {
      states[network] = { exists: false, error: error.message };
    }
  }
  
  return {
    asset_id: assetId,
    timestamp: Date.now(),
    networks: states
  };
}

  'https://bbp-peer-01.netlify.app/peer.js',
  'https://bbp-peer-02.vercel.app/peer.js',
  'https://bbp-peer-03.netlify.app/peer.js',
  'https://bbp-peer-04.workers.dev/peer.js'
];

const BITCOIN_ANCHOR_TXID = 'SEU_TXID_AQUI';

async function loadImageWithGDN(imageId) {
  showLog("[VERIFYING] On-Chain Proof for Fragment_Anchor...");
  
  // 1. Verifica fragmento âncora no Bitcoin
  const anchorValid = await verifyBitcoinAnchor(BITCOIN_ANCHOR_TXID);
  if (anchorValid) {
    showLog(`[VERIFYING] On-Chain Proof... ✓ (TxID: ${BITCOIN_ANCHOR_TXID})`);
  }
  
  showLog("[CONNECTING] BBP Peer Network...");
  
  // 2. Busca fragmentos dos peers
  let fragments = [];
  for (const peer of PEER_NODES) {
    const peerName = peer.split('/')[2];
    showLog(`[CONNECTING] Peer ${peerName}...`);
    
    const peerFragments = await fetchFromPeer(peer, imageId);
    if (peerFragments) {
      fragments.push(...peerFragments);
      showLog(`[DOWNLOADING] ${peerFragments.length} fragments...`);
    }
  }
  
  // 3. Fallback ofuscado
  if (fragments.length < 100) {
    showLog("[INFO] Requesting remaining fragments from backup network...");
    const backup = await fetchFromBackup(imageId);
    fragments.push(...backup);
    showLog(`[DOWNLOADING] ${backup.length} fragments from backup network...`);
  }
  
  showLog("[ASSEMBLING] Image from fragments...");
  return assembleImage(fragments);
}

import CONFIG from './config.js';

async function loadImageWithGDN(imageId) {
  showLog("[VERIFYING] On-Chain Proof for Fragment_Anchor...");
  
  // 1. Verifica fragmento âncora (mock ou real)
  let anchorValid = false;
  if (CONFIG.ENV === 'local') {
    // Mock: verifica se o arquivo existe
    const anchorExists = await fetch(CONFIG.BITCOIN.local.anchorImage).then(r => r.ok);
    anchorValid = anchorExists;
    showLog(`[VERIFYING] On-Chain Proof... ✓ (Mock TxID: mock_anchor_abc123)`);
  } else {
    // Real: consulta mempool.space
    const response = await fetch(`${CONFIG.BITCOIN.production.api}/tx/${ANCHOR_TXID}`);
    anchorValid = response.ok;
    showLog(`[VERIFYING] On-Chain Proof... ✓ (TxID: ${ANCHOR_TXID})`);
  }
  
  showLog("[CONNECTING] BBP Peer Network...");
  
  // 2. Busca fragmentos dos peers (mock ou reais)
  let fragments = [];
  const peers = CONFIG.PEER_NODES[CONFIG.ENV];
  
  for (const peer of peers) {
    const peerName = peer.split('/').pop();
    showLog(`[CONNECTING] Peer ${peerName}...`);
    
    let peerFragments = null;
    if (CONFIG.ENV === 'local') {
      // Mock: importa o arquivo local
      const peerModule = await import(peer);
      peerFragments = await peerModule.getFragment(imageId);
    } else {
      // Real: fetch do peer
      const response = await fetch(peer);
      peerFragments = await response.json();
    }
    
    if (peerFragments) {
      fragments.push(peerFragments);
      showLog(`[DOWNLOADING] ${Object.keys(peerFragments).length} fragments...`);
    }
  }
  
  // 3. Fallback para Arweave (mock ou real)
  if (fragments.length < 100) {
    showLog("[INFO] Requesting remaining fragments from backup network...");
    
    let backupFragments = null;
    if (CONFIG.ENV === 'local') {
      // Mock: busca na pasta local
      const response = await fetch(`${CONFIG.ARWEAVE.local.basePath}${imageId}/fragments.json`);
      backupFragments = await response.json();
    } else {
      // Real: proxy ofuscado
      const response = await fetch(`${CONFIG.PROXY_URL}?id=${imageId}`);
      backupFragments = await response.json();
    }
    
    fragments.push(...backupFragments.fragments);
    showLog(`[DOWNLOADING] ${backupFragments.fragments.length} fragments from backup network...`);
  }
  
  showLog("[ASSEMBLING] Image from fragments...");
  return assembleImage(fragments);
}
